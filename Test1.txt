5.
SELECT COUNT(*) AS Employee_Count 
FROM Worker

6.
SELECT FIRST_NAME, LAST_NAME, SALARY 
FROM Worker 
WHERE SALARY BETWEEN 50000 AND 100000;

7.
SELECT DEPARTMENT, COUNT(WORKER_ID) AS Worker_Count 
FROM Worker 
GROUP BY DEPARTMENT 
ORDER BY Worker_Count DESC;

8.
SELECT SALARY 
FROM Worker W1 
WHERE 4 = (SELECT COUNT(DISTINCT SALARY) FROM Worker W2 WHERE W2.SALARY > W1.SALARY);

9.
SELECT W1.FIRST_NAME, W1.LAST_NAME, W1.SALARY 
FROM Worker W1 
JOIN Worker W2 
ON W1.SALARY = W2.SALARY AND W1.WORKER_ID <> W2.WORKER_ID
ORDER BY W1.SALARY DESC;

10.
SELECT DEPARTMENT 
FROM Worker 
GROUP BY DEPARTMENT 
HAVING COUNT(WORKER_ID) < 3;
